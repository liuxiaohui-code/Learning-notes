# cobra

## import

```go
import "github.com/spf13/cobra" // 命令行工具
```

## simple-start

```go
// server.go
var (
	configPath string

	// ServerCMD command for baas server
	ServerCMD = &cobra.Command{
		Use:   "server",
		Short: "baas server ",
		Run: func(cmd *cobra.Command, args []string) {
			server()
		},
	}
)

func init() {
	ServerCMD.PersistentFlags().StringVarP(&configPath, "config", "c", "", "config path")
}

// main.go
var (
	rootCmd = &cobra.Command{
		Use:   "baas",
		Short: "baas is a manage plantform for hyperledger/fabric",
	}
)

func init() {
	rootCmd.AddCommand(server.ServerCMD)
}
func main(){
    if err := rootCmd.Execute(); err != nil {
		panic(err)
	}
}

// 运行：go run main.go server
```
